Create a new data set with NA replaced by mean value

tnew=t
for (i in seq(1,dim(t)[1])){
  if (is.na(t$steps[i])){
      index=  which(names(meanStepsPer5Min) ==t$interval[i])
      tnew$steps[i]=meanStepsPer5Min[index]
  }
}
